def triangle(a, b, c):
    peri = (a + b + c) / 2
    print("The Semi-perimeter of the Triangle:", peri)
    area = (peri * (peri - a) * (peri - b) * (peri - c)) ** 0.5
    print("The Area of the Triangle is:", area)

def rectangle(a, b):
    peri = 2 * (a + b)
    print("The Perimeter of the Rectangle:", peri)
    area = a * b
    print("The Area of the Rectangle is:", area)

def square(a):
    peri = 4 * a
    print("The Perimeter of the Square:", peri)
    area = a * a
    print("The Area of the Square is:", area)
